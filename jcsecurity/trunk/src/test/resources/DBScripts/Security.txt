1. CREATE TABLE JCP_RESOURCE_TYPE
---------------------------------

 CREATE TABLE JCP_RESOURCE_TYPE
   (RESOURCE_TYPE_ID NUMBER(10,0) NOT NULL ENABLE, 
	RESOURCE_TYPE NUMBER(10,0) NOT NULL ENABLE, 
	RESOURCE_TYPE_NAME VARCHAR2(80 BYTE) NOT NULL ENABLE, 
	RESOURCE_PERMISSION_CLASS VARCHAR2(400 BYTE) NOT NULL ENABLE, 
	CREATED_BY NUMBER(10,0) NOT NULL ENABLE, 
	CREATION_DATE DATE NOT NULL ENABLE, 
	UPDATED_BY NUMBER(10,0) NOT NULL ENABLE, 
	UPDATE_DATE DATE NOT NULL ENABLE, 
	CONSTRAINT JCP_RESOURCE_TYPE_PK PRIMARY KEY (RESOURCE_TYPE_ID) ENABLE, 
	UNIQUE (RESOURCE_TYPE_NAME) ENABLE, 
	UNIQUE (RESOURCE_TYPE) ENABLE
   ) ;

2. CREATE VIEW JCP_PRIVILEGE_PERMISSION_VL
------------------------------------------

   CREATE OR REPLACE FORCE VIEW JCP_PRIVILEGE_PERMISSION_VL 
   (PRINCIPALID, RESOURCEID, PRINCIPALNAME, RESOURCETYPE, RESOURCETYPENAME, 
	RESOURCEPERMISSIONCLASS,RESOURCENAME, RESOURCEACTION) 
	AS SELECT P.PRIVILEGE_ID as principalId, R.RESOURCES_ID as 
	resourceId, P.NAME as principalName, R.RESOURCES_TYPE as resourceType, RT.RESOURCE_TYPE_NAME 
	as resourceTypeName, RT.RESOURCE_PERMISSION_CLASS as resourcePermissionClass, R.RESOURCES_NAME 
	as resourceName, A.NAME as resourceAction FROM JCP_PRIVILEGE P, JCP_RESOURCE_ACTION_PRIVILEGE RAP, 
	JCP_RESOURCEACTION RA, JCP_RESOURCE R, JCP_RESOURCE_TYPE RT, JCP_ACTION A WHERE 
	P.PRIVILEGE_ID = RAP.PRIVILEGE_ID AND RAP.RESOURCE_ACTION_ID = RA.RESOURCE_ACTION_ID AND 
	RA.RESOURCES_ID = R.RESOURCES_ID AND RT.RESOURCE_TYPE = R.RESOURCES_TYPE AND RA.ACTION_ID = A.ACTION_ID;

3. CREATE TABLE JCP_MENU_B 
---------------------------------

CREATE TABLE  JCP_MENU_B ( 
  MENU_ID     NUMBER(10) NOT NULL, 
  DESCRIPTION   VARCHAR2(100)  NOT NULL, 
  PARENT_MENU_ID      NUMBER(10) , 
  STATUS   VARCHAR2(3) NOT NULL, 
  CREATED_BY             NUMBER(10) NOT NULL, 
  CREATION_DATE       DATE        NOT NULL, 
  UPDATED_BY             NUMBER(10) NOT NULL, 
  UPDATE_DATE           DATE       NOT NULL,
  CONSTRAINT   MENU_ID_PK
	PRIMARY KEY (  MENU_ID )
  )

;GO
